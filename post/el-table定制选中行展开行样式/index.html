2. 函数

函数如下所示：

 1methods: {
 2      // 选择全部
 3      handleALL (val) {
 4        this.handleSelect(val);
 5      },
 6      // 选择某一行
 7      handleSelect (row) {
 8        this.selectRow = [];
 9        if (row.length > 0) {
10          row.forEach((value, index) => {
11            this.selectRow.push(value.id);
12          })
13        }
14      },
15      // 选择项发生变化
16      handelSelectChange( val ) {
17        this.handleSelect(val);
18      },
19      // 某一行被展开
20      handleExpand( row ,expandedRows ) {
21        this.expandRow = [];
22        if (expandedRows.length > 0) {
23          expandedRows.forEach((value, index) => {
24            this.expandRow.push(value.id);
25          })
26        }
27      },
28      // 给行绑定类名
29      tableRowClassName({ row, rowIndex }) {
30        let className = '';
31        for(let item of this.selectRow.values()){
32          if(item == row.id){
33            className = 'selected-bgcolor';
34          }
35        }
36        for(let item of this.expandRow.values()){
37          if(item == row.id){
38            className = 'expanded-bgcolor';
39          }
40        }
41        return className;
42      }
43    }
