『…』 扩展运算符能将『数组』转换为逗号分隔的『参数序列』

 1//声明一个数组 ...
 2const tfboys = ['易烊千玺','王源','王俊凯'];
 3// => '易烊千玺','王源','王俊凯'
 4
 5// 声明一个函数
 6function chunwan(){
 7    console.log(arguments);
 8}
 9
10chunwan(...tfboys); // chunwan('易烊千玺','王源','王俊凯')
11
12//1. 数组的合并 情圣  误杀  唐探
13const kuaizi = ['王太利','肖央'];
14const fenghuang = ['曾毅','玲花'];
15//const zuixuanxiaopingguo = kuaizi.concat(fenghuang);
16const zuixuanxiaopingguo = [...kuaizi, ...fenghuang];
17console.log(zuixuanxiaopingguo);
18
19//2. 数组的克隆
20const sanzhihua = ['E','G','M'];
21const sanyecao = [...sanzhihua];//  ['E','G','M']
22console.log(sanyecao);
23
24//3. 将伪数组转为真正的数组
25const divs = document.querySelectorAll('div');
26const divArr = [...divs];
27console.log(divArr); // arguments

ES6 引入 rest 参数，用于获取函数的实参，用来代替 arguments。

 1// ES5 获取实参的方式
 2function date(){
 3	console.log(arguments);
 4}
 5date('白芷','阿娇','思慧');
 6
 7// rest 参数
 8function date(...args){
 9	console.log(args); // filter some every map 
10}
11date('阿娇','柏芝','思慧');
12
13//rest 参数必须要放到参数最后
14function fn(a,b,...args){
15    console.log(a);
16    console.log(b);
17    console.log(args);
18}
19fn(1,2,3,4,5,6);
