1. 数据劫持原理

关于如何实现数据劫持，在Vue的官网有一段描述：

Vue 最独特的特性之一，是其非侵入性的响应式系统。数据模型仅仅是普通的 JavaScript 对象。而当你修改它们时，视图会进行更新。

这使得状态管理非常简单直接，不过理解其工作原理同样重要，这样你可以避开一些常见的问题。

如何追踪变化

当你把一个普通的 JavaScript 对象传入 Vue 实例作为 data 选项，Vue 将遍历此对象所有的 property，并使用 Object.defineProperty 把这些 property 全部转为 getter/setter。

Object.defineProperty 是 ES5 中一个无法 shim 的特性，这也就是 Vue 不支持 IE8 以及更低版本浏览器的原因。

这些 getter/setter 对用户来说是不可见的，但是在内部它们让 Vue 能够追踪依赖，在 property 被访问和修改时通知变更。

每个组件实例都对应一个 watcher 实例，它会在组件渲染的过程中把“接触”过的数据 property 记录为依赖。之后当依赖项的 setter 触发时，会通知 watcher，从而使它关联的组件重新渲染。

存在的问题

我们对所有的property的改变都可以进行监听，但是这里会有2个问题：

  1. 如果一个property不是一个原始变量，而是一个对象，我们就不能对这个对象进行监听了，需要递归的监听下一层；
  2. 如果一个property不是一个原始变量，而是一个数组，我们也不能对这个对象进行监听了；

当然这些问题在Vue3中不存在了，下面我们就来实现一下数据劫持。

2. Vue 初始化

myVue.js：

 1function Vue (options) {
 2    this._init(options);
 3}
 4
 5Vue.prototype._init = function (options) {
 6    var vm = this;
 7    vm.$options = options;
 8    
 9    // 使用data进行state的初始化
10    if(!!vm.$options.data){
11        // 在实例上存储一下data
12        // data有可能是一个函数，也有可能是一个对象
13        vm.$data = 
14            typeof vm.$options.data == 'function' ? 
15            	vm.$options.data.call(vm) 
16        		:
17        		vm.$options.data;
18        
19        // 代理一下，可以使用this.XXX访问data
20        for (let key in vm._data) {
21            Object.defineProperty(vm, key, {
22                get () {
23                    return vm.$data[key];
24                },
25                set (newValue) {
26                    vm.$data[key] = newValue;
27                }
28            })
29        }
30    }
31}

1. 处理对象

使用的是Object.defineProperty，这个方法只有对象才有，如果不是对象是数组，那么就不能使用了。

Observer.js：

 1function Observer (data) {
 2    this.walk(data);
 3}
 4
 5Observer.prototype.walk = function (data) {
 6    Object.keys(data).forEach(key => {
 7        this.defineReactive(data, key, data[key]);
 8    })
 9}
10
11Observer.prototype.defineReactive = function (obj, key, val) {
12    if(typeof val == 'object'){
13        // 递归监听，如果某一项也是对象，那就监听这个对象
14        this.walk(val);
15    } else {
16        Object.defineProperty(obj, key, {
17            get () {
18                // 注意：这里val不可改成obj[key]，会无限递归直至堆栈溢出
19                return val;
20            },
21            set (newValue) {
22                if(val == newValue){
23                    return;
24                } else {
25                    val = newValue;
26                }
27            }
28        })
29    }
30}
31
32
33export default Observer;

2. 重写数组方法

监听数组就不能用 Object.defineProperty了，而是需要重写所有会改变原数组的方法。

首先来回忆一下，可以更改原数组的方法都有哪些？

  1. push和pop
  2. shift和unshift
  3. splice
  4. sort

那么我们就需要重写这些方法，在进行这些操作时，加入我们需要的操作。

Array.js：

 1// 复制一份Array原型，防止污染Array原型的方法
 2const arrayProto = Array.prototype;
 3// 以Array原型为原型，初始化一个一个对象，新对象的__proto__会指向Array.prototype
 4const array = Object.create(arrayProto);
 5
 6// 只需要重写改变原数组的方法，不改变的就不需要重写
 7['push', 'pop', 'shift', 'unshift', 'splice', 'sort', 'reverse'].forEach(funcName => {
 8    // 向我们新建的array对象上添加这些方法
 9    array[funcName] = function (arguments) {
10        let args = Array.from(arguments);
11        arrayProto[funcName].apply(this, args);
12    }
13})
14
15export default array;
3. 监听数组

如果是数组的话，我们肯定是要去监听数组中的每一项：

Observer.js：

 1import customArray form './Array';
 2
 3function Observer (data) {
 4    this.observe(data);
 5}
 6
 7Observer.prototype.observe = function (data) {
 8    // 如果是数组，观察每一项
 9    if(data instanceof Array){
10        this.arrayWalk(data);
11        
12    } elseif (typeof data == 'object') {
13        this.walk(data);
14    }
15}
16
17Observer.prototype.walk = function (data) {
18    Object.keys(data).forEach(key => {
19        this.defineReactive(data, key, data[key]);
20    })
21}
22
23Observer.prototype.defineReactive = function (obj, key, val) {
24    if(val instanceof Array || typeof val == 'object'){
25        // 递归监听，如果某一项也是对象或者数组，那就进入监听它
26        this.observe(val);
27    } else {
28        Object.defineProperty(obj, key, {
29            get () {
30                // 注意：这里val不可改成obj[key]，会无限递归直至堆栈溢出
31                return val;
32            },
33            set (newValue) {
34                if(val == newValue){
35                    return;
36                } else {
37                    val = newValue;
38                }
39            }
40        })
41    }
42}
43
44Observer.prototype.arrayWalk = function (data) {
45    // 实现数组方法的修改
46    data.__proto__ = customArray;
47    data.forEach(item => {
48        this.observe(item);
49    })
50}
51
52
53export default Observer;
4. 监听数组新元素

还有一种情况是，数组新增了元素，那么我们需要对这些新增的元素添加监听，因此我们还需要监听新元素：

Array.js：

 1import Observer from './Observer';
 2
 3// 复制一份Array原型，防止污染Array原型的方法
 4const arrayProto = Array.prototype;
 5// 以Array原型为原型，初始化一个一个对象，新对象的__proto__会指向Array.prototype
 6const array = Object.create(arrayProto);
 7
 8// 只需要重写改变原数组的方法，不改变的就不需要重写
 9['push', 'pop', 'shift', 'unshift', 'splice', 'sort'].forEach(funcName => {
10    // 向我们新建的array对象上添加这些方法
11    array[funcName] = function () {
12        let args = Array.from(arguments);
13        arrayProto[funcName].apply(this, args);
14        // 监听
15        // 会新增数组项的方法只有下面三种
16        let newItemArr;
17        switch (funcName) {
18            case 'push': 
19            case 'unshift':
20                newItemArr = args;
21                break;
22            case 'splice':
23                // 因为splice有三个参数，所以我们要拿第三项
24                newItemArr = args.slice(2);
25                break;
26            default:
27                break;
28        }
29        if(!!newItemArr){
30            new Observer(newItemArr);
31        }
32    }
33})
34
35export default array;
