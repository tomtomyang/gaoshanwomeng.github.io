1.连接数据库

 1// 引入安装的包
 2var mongoose = require('mongoose');
 3mongoose.connect('mongodb://localhost/mongoo');
 4//mongoo这里是代表数据库的名称
 5var db = mongoose.connection;
 6db.on('error', console.error.bind(console, '连接失败'));
 7db.once('open', function() {
 8  console.log('连接成功')
 9});
10// new一个schema：把一个本身非结构化的数据变成结构化
11var kittySchema = new mongoose.Schema({
12    name: String,
13    age:Number,
14    gender:String
15});
16// 根据schema得到一个model，这个model是一个class
17var Kitten = mongoose.model('Kitten', kittySchema);

查询有三种方式find查全部符合要求，findOne查到的第一个，findById通过集合中每条数据特定的_id来查询

查询方式还可以特定条件来查找

2//"$lte"(小于等于)

8// "$nin"查找不为20-24之间的数据
