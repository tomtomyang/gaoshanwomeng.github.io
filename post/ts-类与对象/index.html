其实我们也可以使用get/set封装，而不用继承，就可以达到上面的目的。

 1class Duck {
 2    DuckName: string;
 3    Age: number;
 4    Fly: string;
 5    private _className: string;
 6    constructor(duckName: string, age: number, fly: string) {
 7        this.DuckName = duckName;
 8        this.Age = age;
 9        this.Fly = fly;
10    }
11
12    get className(): string {
13        return this._className;
14    }
15
16    set className(newname: string) {
17        this._className = newname;
18        let outString: string = "我是" + newname + "，我的名字是" + this.DuckName + "，我现在" + this.Age + "岁," + this.Fly;
19        console.log(outString);
20    }
21
22}
23
24
25var getduck = new Duck("黄鸭子", 1, "我会飞");
26getduck.className = "鸭子"; //我是鸭子，我的名字是黄鸭子，我现在1岁,我会飞
27var getduck = new Duck("小黄", 0, "我不会飞");
28getduck.className = "橡皮鸭子"; //我是橡皮鸭子，我的名字是小黄，我现在0岁,我不会飞
