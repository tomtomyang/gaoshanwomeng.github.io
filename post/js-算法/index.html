const arr = [ 31, 23, 34, 2, 13, 134, 1, 23, 456, 52, 31 ];

function quickSort(arr) {
    // 1.找基准数，
    //  比基准数大的放右边（右数组）
    //  比基准数小的放左边（左数组）
    let base_num = arr[0];
    let left_arr = [];
    let right_arr = [];

    // for( let i = 1; i < arr.length ; i++){
    //     if(arr[i] < base_num){
    //         left_arr.push(arr[i]);
    //     }
    //     else {
    //         right_arr.push(arr[i]);
    //     }
    // }

    arr.shift();
    for( let item of arr ) {
        if(item < base_num){
            left_arr.push(item);
        }
        else {
            right_arr.push(item);
        }
    }

    // 2.对左右子数组分别进行快排
    // 条件是子数组中元素个数 >= 2，如果是一个的话肯定是有序的
    if(left_arr.length>=2){
        left_arr = quickSort(left_arr);
    }
    if(right_arr.length>=2){
        right_arr = quickSort(right_arr);
    }

    // 3.合并排序完成的 左数组 基准数 右数组
    return [...left_arr, base_num, ...right_arr];
}

console.log(quickSort(arr));

class Stack {
    constructor(data=[],arrow=-1){
        this.data = data;
        this.arrow = arrow;
    }
    push = ( item )=>{
        this.arrow++;
        this.data[this.arrow] = item;
    }
    pop = ()=>{
        if(this.arrow!=-1){
            let [item] = this.data.splice(this.arrow--);
            return item;
        }
    }
}

class Queue {
    constructor(){
        this.stack1 = new Stack();
        this.stack2 = new Stack();
    }
    push = (item)=> {
        this.stack1.push(item);
    }
    pop = ()=>{
        while(this.stack1.arrow!=-1){
            let item = this.stack1.pop();
            this.stack2.push(item);
        }
        if(this.stack2.arrow!=-1){
            let item = this.stack2.pop();
            return item;
        }
    }
}

/* 
    1只青蛙一次可以跳1级台阶，也可以跳两级台阶，那么跳到n级台阶有几种跳法
    思路：
    1.当我们处于n级台阶时，我们可以是第n-1级台阶跳一步，也可以是n-2级跳两步
    2.那么问题就变成了到达n-1级台阶和n-2级台阶有几种方法
*/

var catche = [1,2];

function jumpStep(n){
    if(typeof catche[n-1] != 'undefined'){
        return catche[n-1];
    }
    else{
        catche[n-1] = jumpStep(n-1)+jumpStep(n-2)
        return catche[n-1];
    }
}

console.log(jumpStep(10));

/* 
    1只青蛙一次可以跳1级台阶，也可以跳两级台阶，还可以跳n级台阶，那么跳到n级台阶有几种跳法
    思路：
    1.当我们处于n级台阶时，我们可以是第n-1级台阶跳一步，也可以是n-2级跳两步
    2.那么问题就变成了到达n-1级台阶和n-2级台阶有几种方法
*/


function jumpStep2(n){
    if(n == 1){
        return 1;
    }
    else{
        return 2*jumpStep2(n-1);
    }
}

console.log(jumpStep2(9));

class linkList {
    constructor(){
        this.head = new Node();
        this.tail = this.head;
    }
    addNode = (itemVal)=>{
        let node = new Node(itemVal)
        this.tail.next = node;
        this.tail = this.tail.next;
    }
}

/*
    输入二叉树的前序遍历和中序遍历，重建出该二叉树
    假设输入的前序遍历和中序遍历的结果中不包含重复的数字

    1. 前序遍历：根 左 右
    2. 中序遍历：左 跟 右

    思路：
    1. 找到根节点：前序遍历的第一个就是根节点
    2. 找到左子树合集：在中序遍历中找到根节点位置，左边就是左子树合集
    3. 找到右子树合集：在中序遍历中找到根节点位置，右边就是右子树合集
    4. 对左右子树进行重排
*/

const pre_traversal = [1,2,4,7,3,5,6,8];
const inorder_traversal = [4,7,2,1,5,3,8,6];

class binaryTreeNode {
    constructor(value, left={}, right={}){
        this.value = value;
        this.left = left;
        this.right = right;
    }
}

function backToBinaryTree(preArr, inorderArr){
    let rootNodeValue = preArr[0];
    let rootNode = new binaryTreeNode(rootNodeValue);

    let rootNodeIndex = inorderArr.indexOf(rootNodeValue);
    let leftInorderArr = inorderArr.slice(0,rootNodeIndex);
    let leftPreArr = preArr.slice(1,leftInorderArr.length+1);
    let rightInorderArr = inorderArr.slice(rootNodeIndex+1);
    let rightPreArr = preArr.slice(leftInorderArr.length+1);

    if(leftInorderArr.length>=1){
        rootNode.left = backToBinaryTree(leftPreArr, leftInorderArr);
    }
    
    if(rightInorderArr.length>=1){
        rootNode.right = backToBinaryTree(rightPreArr, rightInorderArr);
    }

    return rootNode;
}

console.log(backToBinaryTree(pre_traversal,inorder_traversal));

const arr = ['a','b','c','d','#','#','e','#','f','#','#','#','#'];

class Node {
    constructor(val = '', left = null, right = null){
        this.val = val;
        this.left = left;
        this.right = right;
    }
}

function back2Tree(arr){
    if(arr.length!=0){
        let queue = [];
        let nodeVal = arr.shift();
        let rootNode = new Node(nodeVal);
        queue.push(rootNode);
        while(queue.length != 0){
            let head = queue.shift();
            let leftVal = arr.shift();
            let leftNode = new Node(leftVal);
            head.left = leftNode;
            if(leftVal!='#'){
                queue.push(leftNode);
            }
            let rightVal = arr.shift();
            let rightNode = new Node(rightVal);
            head.right = rightNode;
            if(rightVal!='#'){
                queue.push(rightNode);
            }

        }
        return rootNode;
    }  
}

console.log(back2Tree(arr));
