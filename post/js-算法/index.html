2. 用栈实现队列

 1class Stack {
 2    constructor(data=[],arrow=-1){
 3        this.data = data;
 4        this.arrow = arrow;
 5    }
 6    push = ( item )=>{
 7        this.arrow++;
 8        this.data[this.arrow] = item;
 9    }
10    pop = ()=>{
11        if(this.arrow!=-1){
12            let [item] = this.data.splice(this.arrow--);
13            return item;
14        }
15    }
16}
17
18class Queue {
19    constructor(){
20        this.stack1 = new Stack();
21        this.stack2 = new Stack();
22    }
23    push = (item)=> {
24        this.stack1.push(item);
25    }
26    pop = ()=>{
27        while(this.stack1.arrow!=-1){
28            let item = this.stack1.pop();
29            this.stack2.push(item);
30        }
31        if(this.stack2.arrow!=-1){
32            let item = this.stack2.pop();
33            return item;
34        }
35    }
36}

3. 跳台阶

 1/* 
 2    1只青蛙一次可以跳1级台阶，也可以跳两级台阶，那么跳到n级台阶有几种跳法
 3    思路：
 4    1.当我们处于n级台阶时，我们可以是第n-1级台阶跳一步，也可以是n-2级跳两步
 5    2.那么问题就变成了到达n-1级台阶和n-2级台阶有几种方法
 6*/
 7
 8var catche = [1,2];
 9
10function jumpStep(n){
11    if(typeof catche[n-1] != 'undefined'){
12        return catche[n-1];
13    }
14    else{
15        catche[n-1] = jumpStep(n-1)+jumpStep(n-2)
16        return catche[n-1];
17    }
18}
19
20console.log(jumpStep(10));

5. 重建二叉树

 1/*
 2    输入二叉树的前序遍历和中序遍历，重建出该二叉树
 3    假设输入的前序遍历和中序遍历的结果中不包含重复的数字
 4
 5    1. 前序遍历：根 左 右
 6    2. 中序遍历：左 跟 右
 7
 8    思路：
 9    1. 找到根节点：前序遍历的第一个就是根节点
10    2. 找到左子树合集：在中序遍历中找到根节点位置，左边就是左子树合集
11    3. 找到右子树合集：在中序遍历中找到根节点位置，右边就是右子树合集
12    4. 对左右子树进行重排
13*/
14
15const pre_traversal = [1,2,4,7,3,5,6,8];
16const inorder_traversal = [4,7,2,1,5,3,8,6];
17
18class binaryTreeNode {
19    constructor(value, left={}, right={}){
20        this.value = value;
21        this.left = left;
22        this.right = right;
23    }
24}
25
26function backToBinaryTree(preArr, inorderArr){
27    let rootNodeValue = preArr[0];
28    let rootNode = new binaryTreeNode(rootNodeValue);
29
30    let rootNodeIndex = inorderArr.indexOf(rootNodeValue);
31    let leftInorderArr = inorderArr.slice(0,rootNodeIndex);
32    let leftPreArr = preArr.slice(1,leftInorderArr.length+1);
33    let rightInorderArr = inorderArr.slice(rootNodeIndex+1);
34    let rightPreArr = preArr.slice(leftInorderArr.length+1);
35
36    if(leftInorderArr.length>=1){
37        rootNode.left = backToBinaryTree(leftPreArr, leftInorderArr);
38    }
39    
40    if(rightInorderArr.length>=1){
41        rootNode.right = backToBinaryTree(rightPreArr, rightInorderArr);
42    }
43
44    return rootNode;
45}
