class Stack {
    constructor(data=[],arrow=-1){
        this.data = data;
        this.arrow = arrow;
    }
    push = ( item )=>{
        this.arrow++;
        this.data[this.arrow] = item;
    }
    pop = ()=>{
        if(this.arrow!=-1){
            let [item] = this.data.splice(this.arrow--);
            return item;
        }
    }
}

class Queue {
    constructor(){
        this.stack1 = new Stack();
        this.stack2 = new Stack();
    }
    push = (item)=> {
        this.stack1.push(item);
    }
    pop = ()=>{
        while(this.stack1.arrow!=-1){
            let item = this.stack1.pop();
            this.stack2.push(item);
        }
        if(this.stack2.arrow!=-1){
            let item = this.stack2.pop();
            return item;
        }
    }
}

function jumpStep(n){
    if(typeof catche[n-1] != 'undefined'){

const pre_traversal = [1,2,4,7,3,5,6,8];
const inorder_traversal = [4,7,2,1,5,3,8,6];

class binaryTreeNode {
    constructor(value, left={}, right={}){
        this.value = value;
        this.left = left;
        this.right = right;
    }
}

function backToBinaryTree(preArr, inorderArr){
    let rootNodeValue = preArr[0];
    let rootNode = new binaryTreeNode(rootNodeValue);

    let rootNodeIndex = inorderArr.indexOf(rootNodeValue);
    let leftInorderArr = inorderArr.slice(0,rootNodeIndex);
    let leftPreArr = preArr.slice(1,leftInorderArr.length+1);
    let rightInorderArr = inorderArr.slice(rootNodeIndex+1);
    let rightPreArr = preArr.slice(leftInorderArr.length+1);

    if(leftInorderArr.length>=1){
        rootNode.left = backToBinaryTree(leftPreArr, leftInorderArr);
    }
    
    if(rightInorderArr.length>=1){
        rootNode.right = backToBinaryTree(rightPreArr, rightInorderArr);
    }

    return rootNode;
}

const arr = ['a','b','c','d','#','#','e','#','f','#','#','#','#'];

class Node {
    constructor(val = '', left = null, right = null){
        this.val = val;
        this.left = left;
        this.right = right;
    }
}

function back2Tree(arr){
    if(arr.length!=0){
        let queue = [];
        let nodeVal = arr.shift();
        let rootNode = new Node(nodeVal);
        queue.push(rootNode);
        while(queue.length != 0){
            let head = queue.shift();
            let leftVal = arr.shift();
            let leftNode = new Node(leftVal);
            head.left = leftNode;
            if(leftVal!='#'){
                queue.push(leftNode);
            }
            let rightVal = arr.shift();
            let rightNode = new Node(rightVal);
            head.right = rightNode;
            if(rightVal!='#'){
                queue.push(rightNode);
            }

        }
        return rootNode;
    }  
}

console.log(back2Tree(arr));
