3. 使用参数

使用参数时，基本上有两种方式，一种是直接使用这个参数，另一种是通过计算属性，将参数与组件内数据双向绑定。

 1<template>
 2   <el-dialog 
 3   class="dialog-1" 
 4   :title="dialogTitle"
 5   :width="popupWidth"
 6   :visible.sync="visible">
 7    </el-dialog>
 8</template>
 9
10<script>
11export default {
12    name: "myDialog1",
13    props: {
14      // 弹窗标题
15      dialogTitle: {
16        type: String,
17        default: "标题"
18      },
19      // 弹窗隐藏参数
20      centerDialogVisible: {
21        type: Boolean,
22        default() {
23          return false;
24        }
25      },
26      // 弹窗默认宽度
27      popupWidth: {
28        type: String,
29        default() {
30          return "420px";
31        }
32      }
33    },
34    computed: {
35      // 计算弹窗是否隐藏
36      visible: {
37        // 获取初始值centerDialogVisible
38        get() {
39          return this.centerDialogVisible;
40        },
41        // 重新计算visible值
42        set(val) { 
43          this.$emit("updateVisible", val);
44        }
45      }
46    },
47    methods: {
48    }
49}
50</script>
